Hashing vb.net Example

          
          Imports System.Security.Cryptography

          Module Program
            Sub Main()
              Dim input As String = "Hello, world!"
          
              ' Create a function that hashes a string.
              Dim hash As String = HashString(input)
              Console.WriteLine(hash)
          
              ' Create a function that dehashes a string.
              Dim dehashed As String = DehashString(hash)
              Console.WriteLine(dehashed)
            End Sub
          
            ' This function hashes a string using the SHA256 algorithm.
            Function HashString(input As String) As String
              Using hasher As SHA256 = SHA256.Create()
                Dim bytes As Byte() = hasher.ComputeHash(Encoding.UTF8.GetBytes(input))
                Return Convert.ToBase64String(bytes)
              End Using
            End Function
          
            ' This function dehashes a string by reversing the hash operation.
            Function DehashString(input As String) As String
              Using hasher As SHA256 = SHA256.Create()
                Dim bytes As Byte() = Convert.FromBase64String(input)
                Dim dehashed As Byte() = hasher.ComputeHash(bytes)
                Return Encoding.UTF8.GetString(dehashed)
              End Using
            End Function
          End Module
